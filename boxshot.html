import pygame import random # Initialize pygame pygame.init() # Set up the display screen_width = 800 screen_height = 600 screen = pygame.display.set_mode((screen_width, screen_height)) pygame.display.set_caption("Box Shooter") # Define colors black = (0, 0, 0) white = (255, 255, 255) # Set up the player player_width = 50 player_height = 50 player_x = screen_width / 2 - player_width / 2 player_y = screen_height - player_height player_speed = 10 # Set up the boxes box_width = 30 box_height = 30 box_speed = 5 box_list = [] box_timer = 0 box_timer_max = 60 # Set up the bullets bullet_width = 5 bullet_height = 10 bullet_speed = 10 bullet_list = [] # Set up the score score = 0 font = pygame.font.SysFont(None, 30) # Main game loop game_running = True while game_running: # Handle events for event in pygame.event.get(): if event.type == pygame.QUIT: game_running = False elif event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE: bullet_x = player_x + player_width / 2 - bullet_width / 2 bullet_y = player_y bullet_list.append(pygame.Rect(bullet_x, bullet_y, bullet_width, bullet_height)) # Move the player keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] and player_x > 0: player_x -= player_speed elif keys[pygame.K_RIGHT] and player_x < screen_width - player_width: player_x += player_speed # Move the boxes for box in box_list: box.y += box_speed # Spawn new boxes if box_timer < box_timer_max: box_timer += 1 else: box_x = random.randint(0, screen_width - box_width) box_y = -box_height box_list.append(pygame.Rect(box_x, box_y, box_width, box_height)) box_timer = 0 # Move the bullets for bullet in bullet_list: bullet.y -= bullet_speed # Check for collisions between bullets and boxes for bullet in bullet_list: for box in box_list: if bullet.colliderect(box): box_list.remove(box) bullet_list.remove(bullet) score += 10 # Draw everything screen.fill(black) pygame.draw.rect(screen, white, (player_x, player_y, player_width, player_height)) for box in box_list: pygame.draw.rect(screen, white, box) for bullet in bullet_list: pygame.draw.rect(screen, white, bullet) score_text = font.render("Score: {}".format(score), True, white) screen.blit(score_text, (10, 10)) pygame.display.update() # Clean up pygame.quit()
